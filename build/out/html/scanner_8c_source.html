805 /*
806  Convert the given keyword string to a token. If it is not a keyword,
807  then SYMBOL_TOKEN is returned. Note that this does not convert non-
808  keywords to a token. Non-keywords cause SYMBOL_TOKEN to be returned
809  as well.
810 */
811 token_t str_to_token(const char* str) {
812 
813  // simple binary search
814  token_t retv = SYMBOL_TOKEN;
815  int start = 0, end = TOKEN_MAP_SIZE - 1;
816  int mid = (start + end) / 2;
817 
818  while(start <= end) {
819  int spot = strcmp(str, token_map[mid].str);
820  if(spot > 0)
821  start = mid + 1;
822  else if(spot < 0)
823  end = mid - 1;
824  else {
825  retv = token_map[mid].tok;
826  break;
827  }
828 
829  mid = (start + end) / 2;
830  }
831 
832  return retv;
833 }

851 /*
852  Returns the token in the token structure provided.
853 */
854 token_t get_tok() {
855 
856  int ch, finished = 0;
857  token_t tok = NONE_TOKEN;
858 
859  skip_ws();
860  init_char_buffer(scanner_buffer);
861 
862  while(!finished) {
863  ch = get_char();
864  switch(ch) {
865  case END_FILE:
866  file_flag++;
867  tok = END_OF_FILE;
868  finished++;
869  break;
870  case END_INPUT:
871  tok = END_OF_INPUT;
872  finished ++;
873  break;
874  case '\"': // read a double quoted string
875  tok = read_dquote();
876  finished ++;
877  break;
878  case '\'': // read a single quoted string
879  tok = read_squote();
880  finished ++;
881  break;
882  case '/': // may have a comment or an operator
883  tok = comment_or_operator();
884  skip_ws();
885  if(tok != NONE_TOKEN)
886  finished ++;
887  // continue if it was a comment
888  break;
889  default:
890  if(ch == END_FILE)
891  unget_char(ch);
892  else if(isdigit(ch)) { // beginning of a number. Could be hex, dec, or float
893  unget_char(ch);
894  tok = read_number_top();
895  if(tok != NONE_TOKEN)
896  finished ++;
897  }
898  else if(isalpha(ch)) { // could be a symbol or a keyword
899  unget_char(ch);
900  tok = read_word();
901  if(tok != NONE_TOKEN)
902  finished++;
903  }
904  else if(ispunct(ch)) { // some kind of operator (but not a '/' or a quote)
905  tok = read_punct(ch);
906  if(tok != NONE_TOKEN)
907  finished ++;
908  }
909  else {
910  warning("Unknown character, ignoring. (0x%02X) (%d) \'%c\'", ch, ch, ch);
911  }
912  }
913  }
914  return tok;
915 }

917 /*
918  Retrieve a token and compare it against a token array. If the received
919  token is not in the array, then issue a syntax error and return the
920  ERROR_TOKEN. The array must be terminated with the NONE_TOKEN. (AKA 0)
921 */

923 
924  token_t token = get_tok();
925  for(int i = 0; arr[i] != NONE_TOKEN; i++) {
926  if(token == arr[i])
927  return token;
928  }

934  get_file_name(), get_line_no(), get_column_no());
935  for(int i = 0; arr[i] != NONE_TOKEN; i++) {
936  fprintf(fp, "%s", token_to_str(arr[i]));

938  fprintf(fp, ",");
939  fprintf(fp, " ");

942  return ERROR_TOKEN;
943 }
