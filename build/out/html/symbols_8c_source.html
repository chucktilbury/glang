1 /*
2  Symbols are stored in a hash table. All symbols are decorated to represent
3  the type attributes of the symbol. When a symbol is looked up, it gets
4  undecorated so it can be found in the table. Names are decorated with a
5  preceeding '$' and type parameters are decorated with a preceeding '@'.
6 
7  for example a class method defined in a class named some_cls as
8  public void some_meth(int, string) is decorated as
9  $some_cls$some_meth@int@string
10 
11  If that method defines an int variable named 'var', then that is stored as
12  $some_cls$some_meth@int@string$var
13  The type information is kept in the actual symbol table entry.
14 
15  If a symbol is defined in an anonymous block, such as a while loop, it is
16  assigned a serial number according to the number of '{' that have been seen
17  such as
18  $some_cls$some_meth@int@string$0003$var
19 
20  When the above defined method is referenced in the source code, it will be
21  seen as some_cls.some_meth(var1, var2). The resolver searches out the types
22  of the parameters and any assignment to recreate the fully decorated name.
23 
24  When the symbol is rendered into C code, the '$' and '@' characters are
25  converted to '_'.
26 
27  */

239 
240  char_buffer_t buf = create_char_buffer();
241  const char* tpt = name;
242  int finished = 0;
243  int state = 0;
244 
245 
246  while(!finished) {

251  add_char_buffer(buf, '.');
252  else if(*tpt == '@') {
253  add_char_buffer(buf, '(');
254  state = 1;
255  }
256  else
257  add_char_buffer(buf, *tpt);
258  break;
259  case 1:
260  if(*tpt == '@')
261  add_char_buffer(buf, ',');
262  else if(*tpt == 0) {
263  add_char_buffer(buf, ')');
264  finished++;
265  }
266  else {
267  add_char_buffer(buf, *tpt);
268  }
269  break;
270  }
271  }
272  tpt = get_char_buffer(buf);
273  destroy_char_buffer(buf);
274  return STRDUP(tpt);
